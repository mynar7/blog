```js react-live use-render
function banana() {
    const h2 = document.getElementById('magic-h2')
    h2.textContent = "banana"
}

render(
    <div>
        <h2 id="magic-h2">~Magical~</h2>
        <button onClick={banana}>Click me!</button>
    </div>
)
```

```js react-live
function Counter() {
    const [count, setCount] = React.useState(0)
    function countUp(count) {
        const newCount = count + 1
        setCount(newCount)
    }
    return (
        <div>
            <h2>Count is {count}</h2>
            <button onClick={() => countUp(count)}>Increment</button>
        </div>
    )
}
```
